template< class Key, class Hash, class KeyEqual, class Allocator >

void swap( unordered_multiset<Key,Hash,KeyEqual,Allocator> &lhs,

           unordered_multiset<Key,Hash,KeyEqual,Allocator> &rhs );
 (since C++11)

Specializes the std::swap algorithm for std::unordered_multiset. Swaps the contents of lhs and rhs. Calls lhs.swap(rhs).
